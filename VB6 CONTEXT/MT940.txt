Public Sub ADO_Connect()

On Error GoTo Error_found

    Dim strParamFile2 As String
    Dim strConn As String
    Dim iCount As Integer
    Dim fileno As Integer
    
    fileno = FreeFile()
    'HOME: pinalitan ang filename, dinagdagan ng 2
    strParamFile2 = App.path & "\casarepconn.txt"
    
    Open strParamFile2 For Input As #fileno
    
    Line Input #fileno, strTextLine
        
    iCount = 0
    If strTextLine = "[CONNSTRING]" Then
        While Not EOF(fileno)
            Line Input #fileno, strTextLine
            iCount = iCount + 1
            
            Select Case iCount
                Case 1:
                    strConn = strTextLine
            End Select
            
            Wend
    End If
    
    Close #fileno
    
    Set conn = New ADODB.connection
    conn.ConnectionString = strConn
    conn.Open
    
Error_found:
If err.number <> 0 Then MsgBox err.Description, vbExclamation, "ERROR"

End Sub

Private Function getCurrencyCode(code As String) As String

'used for returning currencycode depending on the Code Passed.
'16 Euro
'17 JPY
'18 CNY
'19 USD
'01,02,03,11,12,13,31,32  - PHP

Select Case code
    Case "16":
        getCurrencyCode = "EUR"
        
    Case "17":
        getCurrencyCode = "JPY"
    
    Case "18":
        getCurrencyCode = "CNY"
    
    Case "19":
        getCurrencyCode = "USD"
    
    Case Else:
        getCurrencyCode = "PHP"
End Select


End Function
